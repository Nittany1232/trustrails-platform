JavaScript API Methods
JavaScript API Usage
const widget = document.querySelector('trustrails-widget');

// Check if user session is ready
if (widget.isUserReady()) {
  console.log('User ID:', widget.getUserId());
  console.log('User Email:', widget.getUserEmail());
  console.log('Full Session:', widget.getUserSession());
}

// Refresh user session if needed
await widget.refreshUserSession();

// Listen for user session events
widget.addEventListener('trustrails-user-ready', (event) => {
  console.log('User session ready:', event.detail);
});
